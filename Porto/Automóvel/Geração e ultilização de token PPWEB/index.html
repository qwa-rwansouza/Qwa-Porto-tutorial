Geração de token e utilização

public class TesteToken {
	public static void main(String[] args) {
		
		List<NumeroOpcaoCalculoEnum> opcoesCalculo = Lists.newArrayList(NumeroOpcaoCalculoEnum.PRIMEIRA_OPCAO_CALCULO);
		
		String token = TokenPPWImpressaoBuilder.novo()
				//Código Documento
		   .comNumeroDocumeto(104489481)
		   .comOpcoesCalculo(opcoesCalculo)
				/*
					Tipo Documento (não copie e cole direto  veja que é passado como ENUM)
				*/
		   .comTiposRelatorio(
				   TipoDocumentoImpressaoControlePPWeb.ORCAMENTO_AZUL, 
				   TipoDocumentoImpressaoControlePPWeb.PROPOSTA_AZUL, 
				   TipoDocumentoImpressaoControlePPWeb.PERFIL_AZUL, 
				   TipoDocumentoImpressaoControlePPWeb.VANTAGENS_AZUL, 
				   TipoDocumentoImpressaoControlePPWeb.AUTORIZACAO_DEBITO_EM_CONTA, 
				   TipoDocumentoImpressaoControlePPWeb.CARTA_COMPARATIVA)
		   .comPerfilAtendente(false)
		   .criar();
		
		System.out.println(token);
	}
}
